\section{Project Overview}
% =====================================================================

\begin{frame}{What is TeacherLM?}
  \begin{block}{One-line definition}
    A full-stack, \textbf{multi-agent RAG} tutoring platform that turns
    uploaded course documents into a personal AI teacher --- every answer
    is grounded \textit{exclusively} in the uploaded files, never the LLM's
    training data.
  \end{block}

  \vspace{0.2cm}
  \textbf{What a student can do:}
  \begin{itemize}
    \setlength\itemsep{1pt}
    \item \textbf{Upload} course files (PDFs, slides, documents).
    \item \textbf{Chat} with a warm, encouraging AI teacher.
    \item \textbf{Generate} quizzes, mind maps, podcasts from the materials.
    \item \textbf{Track progress} per-concept, automatically.
  \end{itemize}

  \vspace{0.2cm}
  \begin{exampleblock}{Why this matters}
    Generic chatbots hallucinate, ignore the actual course context, and
    don't adapt to the student. TeacherLM enforces \textbf{groundedness}
    and \textbf{adaptive teaching} as first-class system properties.
  \end{exampleblock}
\end{frame}

\begin{frame}{The Four Core Principles}
  \begin{tcolorbox}[colback=eduLightBlue, colframe=eduBlue, title=\textbf{1. Strict Grounding}]
    Every factual claim must trace back to a retrieved chunk. If the
    material doesn't cover it, the teacher politely refuses --- it never
    falls back to training data.
  \end{tcolorbox}

  \begin{tcolorbox}[colback=eduLightGreen, colframe=eduGreen, title=\textbf{2. Learner-Aware}]
    A per-conversation \texttt{LearnerState} tracks understood vs.
    struggling concepts, mastery scores, and session turns.
    Every generator receives this state and adapts.
  \end{tcolorbox}

  \begin{tcolorbox}[colback=eduPurple!10, colframe=eduPurple, title=\textbf{3. Multi-Agent / Plugin Architecture}]
    Each output type is an independent FastAPI microservice (a
    \textit{generator}) registered in a JSON manifest. The backend is a
    thin orchestrator. Adding an agent = one new service + one JSON entry.
  \end{tcolorbox}

  \begin{tcolorbox}[colback=eduLightOrange, colframe=eduOrange, title=\textbf{4. Fully Local LLM}]
    All inference runs through \textbf{Ollama} on the host machine ---
    no OpenAI, no cloud APIs. Course materials never leave the box.
  \end{tcolorbox}
\end{frame}

\begin{frame}{Data Flow for a Single Chat Message}
  \scriptsize
  \begin{enumerate}
    \item Frontend $\xrightarrow{\textsf{POST /api/conversations/\{id\}/chat}}$ Backend.
    \item Backend persists the user turn in \textbf{PostgreSQL} (\texttt{Message} table).
    \item \textbf{Retrieval Orchestrator} picks a mode (e.g.\ \texttt{semantic\_topk} for chat)
          and queries \textbf{Qdrant} via hybrid (dense + BM25) retrieval, fused with RRF.
    \item Optional \textbf{cross-encoder reranking} (\texttt{bge-reranker-base}).
    \item Backend loads the \textbf{LearnerState} and assembles a \texttt{GeneratorInput}.
    \item The \textbf{Dispatcher} resolves the target generator via the JSON registry and
          calls its \texttt{/run} endpoint (HTTP + SSE).
    \item The generator runs its multi-step pipeline and \textbf{streams SSE events}:
          \texttt{analysis} $\to$ \texttt{sources} $\to$ \texttt{token}* $\to$ \texttt{done}.
    \item Backend proxies the SSE stream back to the frontend and persists the
          assistant turn + \textbf{learner state updates} on completion.
  \end{enumerate}
\end{frame}

\begin{frame}{Step 1--4: Ingestion (run on background workers)}
  \scriptsize
  \begin{block}{1. Upload}
    File persisted in \textbf{MinIO}, row inserted in PostgreSQL with status
    \texttt{uploaded}. An \textbf{arq} job is enqueued in Redis.
  \end{block}

  \begin{block}{2. Parse}
    The worker streams raw bytes to \textbf{LlamaCloud} ($\geq 1.0$),
    which returns clean Markdown. The Markdown is written back to MinIO.
    \textit{Why LlamaCloud?} PDF layouts (multi-column, tables, equations,
    headers/footers) defeat local PDF extractors --- noisy Markdown
    degrades chunking and retrieval downstream.
  \end{block}

  \begin{block}{3. Cleaning + Structure extraction}
    \texttt{DocumentCleaningService} strips noise (logos, page numbers).
    \texttt{CourseStructureExtractor} produces a \texttt{CourseDocument}
    with hierarchical \texttt{CourseSection}s (heading paths, summaries,
    equations, tables, timeline events, key concepts).
  \end{block}

  \begin{block}{4. Chunking --- structured, not naive}
    \texttt{ChunkingService.chunk\_course\_document} walks each section:
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item Split section text by paragraphs $\to$ sentences (regex on
            \texttt{[.!?]} + sentence-start lookahead). Preserves structured blocks (tables, lists, \texttt{\$\$...\$\$}).
      \item Greedy-merge sentences into chunks targeting
            \texttt{chunk\_max\_tokens = 512} with
            \texttt{chunk\_overlap\_tokens = 50}.
      \item Token approximation: $\mathrm{tokens}(t) \approx 1.3 \cdot |\mathrm{words}(t)|$ --- no tokenizer load.
      \item Each chunk carries metadata: \texttt{document\_id},
            \texttt{section\_id}, \texttt{heading\_path},
            \texttt{section\_summary}, \texttt{key\_concepts},
            \texttt{prev\_/next\_chunk\_id} for neighbor expansion.
    \end{itemize}
  \end{block}
\end{frame}

\section{Hybrid Retrieval and RRF}
% =====================================================================

\begin{frame}{Hybrid Retrieval --- Two Signals, One Ranking}
  \centering
  \begin{tikzpicture}[node distance=0.5cm and 0.6cm]
    \node[svc, fill=eduLightBlue, minimum width=3cm] (q) {\textbf{User query}};

    \node[svcgreen, below left=1cm and 0.5cm of q, minimum width=3.2cm]
      (dense) {\textbf{Dense} (fastembed)\\Qdrant top-K=20};
    \node[svcorange, below right=1cm and 0.5cm of q, minimum width=3.2cm]
      (sparse){\textbf{Sparse / BM25}\\(rank-bm25 Okapi)};

    \node[svcpurple, below=1.5cm of q, minimum width=4.5cm] (rrf)
      {\textbf{Reciprocal Rank Fusion}\\$K=60$, parameter-free};

    \node[svc, fill=eduRed!15, draw=eduRed, below=of rrf, minimum width=4.5cm] (out)
      {\textbf{Fused top-K chunks}};

    \draw[arr] (q.south) -- (dense.north);
    \draw[arr] (q.south) -- (sparse.north);
    \draw[arrG] (dense.south) -- (rrf.north -| dense);
    \draw[arrG] (sparse.south) -- (rrf.north -| sparse);
    \draw[arrG] (rrf) -- (out);
  \end{tikzpicture}

  \vspace{0.1cm}
  \scriptsize
  \textbf{Why both?} Dense models capture \textit{semantics} (``causes of
  photosynthesis'' $\sim$ ``light-dependent reactions''); BM25 captures
  \textit{exact tokens} (Henderson--Hasselbalch, RMSE, nDCG). The fusion
  ensures both are surfaced --- and it's tuning-free.
\end{frame}

\begin{frame}{Reciprocal Rank Fusion --- The Math}
  \begin{block}{Definition}
    For each candidate document $d$ that appears in any ranking list
    $L_i$ at rank $r_i(d)$ (counting from 1):
    \[
      \mathrm{score}(d) \;=\; \sum_{i=1}^{n} \frac{1}{K + r_i(d)},
      \qquad K = 60.
    \]
  \end{block}

  \begin{itemize}
    \setlength\itemsep{1pt}
    \item \textbf{Parameter-free} (the constant $K=60$ comes from the
          original Cormack et al.\ paper and is robust in practice).
    \item Does \textbf{not} require score normalisation across rankers ---
          BM25 scores and cosine similarities live on different scales.
    \item Documents not present in a list contribute $0$ from that list.
    \item Implemented in \texttt{teacherlm\_core/retrieval/hybrid\_retriever.py}
          as \texttt{\_rrf\_fuse(rankings, top\_k)}.
  \end{itemize}

  \begin{exampleblock}{Tiny worked example}
    \scriptsize
    Doc $A$ at rank 1 in Dense, rank 5 in BM25:
    $\frac{1}{61} + \frac{1}{65} \approx 0.0317$.\\
    Doc $B$ at rank 2 in Dense, missing in BM25:
    $\frac{1}{62} \approx 0.0161$.\\
    Doc $C$ rank 10 in both: $2/70 \approx 0.0286$. $\Rightarrow A > C > B$.
  \end{exampleblock}
\end{frame}

\begin{frame}[fragile]{HybridRetriever --- Implementation Sketch}
\begin{lstlisting}[language=Python]
class HybridRetriever:
    async def retrieve(self, query: str, top_k: int = 20) -> list[Chunk]:
        dense_hits  = await self._dense_search(query, limit=self.dense_top_k)
        sparse_hits = self._sparse_search(query, limit=self.sparse_top_k)
        return _rrf_fuse([dense_hits, sparse_hits], top_k=top_k)

def _rrf_fuse(rankings: list[list[Chunk]], top_k: int) -> list[Chunk]:
    fused: dict[str, float] = {}
    by_id: dict[str, Chunk] = {}
    for ranking in rankings:
        for rank, chunk in enumerate(ranking):
            fused[chunk.chunk_id] = (
                fused.get(chunk.chunk_id, 0.0) + 1.0 / (RRF_K + rank + 1)
            )
            by_id.setdefault(chunk.chunk_id, chunk)
    ordered = sorted(fused.items(), key=lambda kv: kv[1], reverse=True)
    return [Chunk(..., score=score) for cid, score in ordered[:top_k]]
\end{lstlisting}
\scriptsize
Each fused chunk's \texttt{score} carries the RRF score; the original
dense / BM25 magnitudes are discarded after fusion.
\end{frame}

\begin{frame}{Cross-Encoder Reranking}
  \begin{block}{The two-stage retrieval pattern}
    \scriptsize
    \begin{itemize}
      \setlength\itemsep{1pt}
      \item \textbf{Stage 1 (recall)}: bi-encoder embeddings + BM25 fetch
            $\sim$20--30 candidates quickly --- they're cheap because the
            query and chunks are encoded independently.
      \item \textbf{Stage 2 (precision)}: a \textbf{cross-encoder}
            (\texttt{BAAI/bge-reranker-base}) scores each
            \texttt{(query, chunk)} pair \textit{jointly}, attending across
            the two texts. Much slower per pair, but vastly more accurate.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Reranker code (\texttt{teacherlm\_core/retrieval/reranker.py})}
    \tiny
\begin{verbatim}
class CrossEncoderReranker:
    def __init__(self, model_name="BAAI/bge-reranker-base"):
        from fastembed.rerank.cross_encoder import TextCrossEncoder
        self._encoder = TextCrossEncoder(model_name=model_name)

    async def rerank(self, query, chunks, top_k=5):
        scores = await asyncio.to_thread(
            self._score, query, [c.text for c in chunks]
        )
        ranked = sorted(zip(scores, chunks), reverse=True)[:top_k]
        return [Chunk(text=c.text, source=c.source, score=float(s),
                      chunk_id=c.chunk_id, metadata=c.metadata)
                for s, c in ranked]
\end{verbatim}
  \end{exampleblock}

  \scriptsize
  The backend's \texttt{RetrievalOrchestrator} also supports
  \textit{group-aware} reranking for comparison queries: it splits hits
  by \texttt{matched\_query\_term} metadata and round-robins ranked
  per-term groups so each term keeps representation.
\end{frame}

\section{The Five Retrieval Modes}
% =====================================================================

\begin{frame}{One Retriever, Five Strategies}
  \scriptsize
  Each output type has different information needs --- the
  \texttt{RetrievalOrchestrator} dispatches on \texttt{output\_type}:

  \vspace{0.1cm}
  \renewcommand{\arraystretch}{1.3}
  \begin{tabular}{p{2.0cm} p{2.4cm} p{8.5cm}}
    \toprule
    \rowcolor{eduBlue!20}\textbf{Output type} & \textbf{Mode} & \textbf{Strategy} \\
    \midrule
    \texttt{text} (chat)            & \texttt{semantic\_topk}     & Top-K closest to query, after hybrid + rerank. \\
    \texttt{quiz}                   & \texttt{coverage\_broad}    & MMR (Jaccard-token) --- relevance vs.\ diversity. \\
    \texttt{report}, \texttt{mindmap}, \texttt{presentation}
                                    & \texttt{topic\_clusters}    & K-means on chunk embeddings; one representative per cluster. \\
    \texttt{podcast}                & \texttt{narrative\_arc}     & Intro-like chunk + relevant middle + conclusion-like chunk. \\
    \texttt{chart}                  & \texttt{relationship\_dense}& Ranks by entity + verb regex density. \\
    \bottomrule
  \end{tabular}

  \vspace{0.2cm}
  \begin{block}{Why this matters}
    Chat needs the \textbf{single most relevant} answer. A quiz needs
    \textbf{breadth} so questions don't repeat. A podcast needs
    \textbf{narrative shape}. A chart needs chunks dense in
    \textbf{entities and relations}. Same vectors, five lenses.
  \end{block}
\end{frame}

\begin{frame}{Mode 1: \texttt{semantic\_topk} (Chat)}
  \scriptsize
  Plain hybrid retrieval over the conversation's collection, capped to
  \texttt{retrieval\_top\_k}.

  \begin{exampleblock}{Backend chat-overview special-case}
    \scriptsize
    If the query looks like ``what is this course about?'' / ``r\'esume
    le cours'' the orchestrator skips top-K and instead returns:
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item the \textbf{course study guide} chunk (if present), plus
      \item \textbf{representative} chunks across the corpus, plus
      \item the \textbf{full course outline}.
    \end{itemize}
    Intent + course-target keywords are detected with multilingual sets
    (EN / FR / Arabic) and phrase regexes.
  \end{exampleblock}
\end{frame}

\begin{frame}{Mode 2: \texttt{coverage\_broad} (Quizzes) --- MMR}
  \begin{block}{Maximal Marginal Relevance}
    \scriptsize
    Pool a wider candidate set, then greedily pick chunks balancing
    \textbf{query relevance} against \textbf{diversity} from already-picked:
    \[
      \mathrm{MMR}(c) \;=\; \lambda \, \mathrm{rel}(c, q) \;-\; (1-\lambda)\,
      \max_{c' \in S} \mathrm{sim}(c, c'),
    \]
    where $\lambda = 0.7$ (relevance bias). Similarities are
    \textbf{Jaccard} over BM25 tokens.
  \end{block}

  \scriptsize
  \textbf{Why for quizzes?} A naive top-K returns 20 near-duplicates of
  the same paragraph. MMR forces the question generator to see
  \textbf{different facets} of the material --- the quiz then spans more
  concepts and Bloom levels.
\end{frame}

\begin{frame}{Mode 3: \texttt{narrative\_arc} (Podcasts)}
  \scriptsize
  Heuristically picks chunks shaped like a podcast story:
  \begin{itemize}
    \setlength\itemsep{0pt}
    \item \textbf{Intro}: 2 chunks containing keywords
          \texttt{introduction / overview / preface / abstract}, falling back
          to the first 10\,\% of the document.
    \item \textbf{Middle}: hybrid retrieval (top-6 against the query).
    \item \textbf{Conclusion}: 2 chunks containing
          \texttt{conclusion / summary / in summary / final thoughts}, falling
          back to the last 10\,\%.
  \end{itemize}
  De-duped by \texttt{chunk\_id}. The podcast generator then has a real
  beginning / middle / end to script around.
\end{frame}

\begin{frame}{Mode 4: \texttt{topic\_clusters} (Mind maps, Reports)}
  \begin{block}{Algorithm}
    \scriptsize
    \begin{enumerate}
      \setlength\itemsep{0pt}
      \item Hybrid-retrieve a wide pool ($\geq 30$ candidates).
      \item Re-embed the pool with fastembed.
      \item Run K-means (\texttt{n\_clusters}, default 6, max 20 iters).
      \item For each cluster, return the chunk \textbf{closest to the centroid}.
    \end{enumerate}
  \end{block}

  \scriptsize
  \textbf{Why for mind maps?} A mind map needs a balanced spread of
  themes --- not 6 chunks all about ``Android intents''. K-means
  partitions the semantic space; centroid chunks are the most
  representative anchors.
\end{frame}

\begin{frame}{Mode 5: \texttt{relationship\_dense} (Charts)}
  \begin{block}{Density heuristic}
    \scriptsize
    For each candidate chunk:
    \[
      \mathrm{density}(c) \;=\; \frac{\#\textsf{entities}(c) + \#\textsf{verbs}(c)}{\#\textsf{tokens}(c)}.
    \]
    Entities $\sim$ \texttt{\textbackslash b(?:[A-Z][a-z]+)(?:\textbackslash s+[A-Z][a-z]+)+\textbackslash b}\\
    Verbs $\sim$ \texttt{\textbackslash b\textbackslash w+(?:ed|ing|es)\textbackslash b}
  \end{block}

  \scriptsize
  Chunks dense in proper-noun pairs and action verbs make better
  inputs for entity-relation diagrams than narrative prose. Chart
  generation is wired to this mode in \texttt{generators\_registry.json}
  but the generator itself is on the roadmap.
\end{frame}

\begin{frame}{Teacher Generator --- Step 1: Query Analysis}
  \scriptsize
  \texttt{services/query\_analyzer.py} calls
  \texttt{generate\_structured[QueryAnalysis](\dots)} which uses
  Ollama's \texttt{format=} parameter with the Pydantic JSON schema.
  No LangChain --- just \texttt{ollama.AsyncClient} + automatic
  retry + self-repair on validation failure.

  \begin{block}{\texttt{QueryAnalysis} schema (Pydantic V2)}
    \scriptsize
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item \texttt{intent}: one of \{\texttt{ask\_explanation},
            \texttt{ask\_definition}, \texttt{ask\_example},
            \texttt{seek\_confirmation}, \texttt{express\_confusion},
            \texttt{demonstrate\_understanding}, \texttt{off\_topic}\}.
      \item \texttt{confusion\_level}: \texttt{low / medium / high}.
      \item \texttt{targets\_concept}: short concept string or null.
    \end{itemize}
  \end{block}

  This structured output \emph{drives the rest of the pipeline}: mode
  selection, the optional off-topic refusal, the choice of system
  prompt, and the eventual \texttt{LearnerUpdates}.
\end{frame}

\begin{frame}{Teacher Generator --- Step 2: Response Mode}
  \scriptsize
  \texttt{services/response\_mode.py} is a rule-based decision tree
  over the analysis + \texttt{LearnerState}. It selects one of four
  mode-specific system prompts:

  \begin{tcolorbox}[colback=eduLightBlue, colframe=eduBlue, title=\texttt{explain}]
    Direct explanation from the materials. Default for fresh
    questions or ``stuck'' students (high \texttt{turns\_since\_progress}).
  \end{tcolorbox}
  \begin{tcolorbox}[colback=eduLightGreen, colframe=eduGreen, title=\texttt{guide} (Socratic)]
    The teacher asks leading questions instead of giving the answer
    directly --- triggered when \texttt{confusion\_level == "medium"} or
    the student is exploring.
  \end{tcolorbox}
  \begin{tcolorbox}[colback=eduLightOrange, colframe=eduOrange, title=\texttt{quiz\_back}]
    The teacher tests the student's understanding before confirming
    --- triggered when intent is \texttt{seek\_confirmation} but mastery
    on the target concept is unclear.
  \end{tcolorbox}
  \begin{tcolorbox}[colback=eduPurple!10, colframe=eduPurple, title=\texttt{affirm}]
    Validates correct understanding and advances. Triggered when the
    student \texttt{demonstrates\_understanding} on a concept above the
    mastery threshold.
  \end{tcolorbox}
\end{frame}

\begin{frame}{Teacher Generator --- Steps 3--4: Guards \& Special Cases}
  \scriptsize
  \begin{block}{Off-topic guard}
    Before any LLM call, the pipeline checks
    \texttt{ranked\_chunks[0].score < min\_relevance\_score}. If true,
    it streams a polite refusal explaining the question isn't covered
    by the uploaded files and returns. \textbf{This is a hard
    anti-hallucination gate.}
  \end{block}

  \begin{block}{Special-case handlers (deterministic before LLM)}
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item \texttt{\_is\_formula\_question(\dots)} --- if the user
            asks about a formula / equation / RMSE / nDCG, the pipeline
            extracts \texttt{\$\$\dots\$\$} blocks and symbol definitions
            via regex from chunks, builds a deterministic
            \textbf{formula card} (formula + symbol meanings + plain-language
            note), and returns it with confidence = 1.0.
      \item \texttt{\_is\_course\_overview\_question(\dots)} --- detects
            ``what is this course about?'' (multilingual). Uses the
            study-guide chunk if present, else extracts modules via
            \texttt{\textasciicircum Module \textbackslash d+:} regex.
            Returns a structured Big Picture / Main Modules / How To Study
            Markdown response.
    \end{itemize}
  \end{block}
  These short-circuits give crisp, source-citing answers when LLM
  generation would risk paraphrase noise.
\end{frame}

\begin{frame}{Teacher Generator --- Steps 5--6: Generation \& Scoring}
  \scriptsize
  \textbf{Streaming reply.}
  \texttt{llm.stream\_reply(system, chat\_history, user\_message)} wraps
  \texttt{ollama.AsyncClient.chat(stream=True)} and yields token deltas.
  Each delta is emitted as an SSE \texttt{token} event, then
  concatenated into the full response.

  \textbf{Confidence scoring (\texttt{teacherlm\_core.confidence}).}
  \begin{itemize}
    \setlength\itemsep{0pt}
    \item \textbf{Groundedness}: mean over sentences of
          \[
            g(s) \;=\; \frac{|\textsf{tokens}(s) \setminus \textsf{stop} \,\cap\, \bigcup_c \textsf{tokens}(c)|}{|\textsf{tokens}(s) \setminus \textsf{stop}|}.
          \]
          Near 1.0 $\Rightarrow$ well-grounded; near 0 $\Rightarrow$ possible
          hallucination.
    \item \textbf{Coverage}: fraction of distinct \textit{query} content
          tokens that appear across the retrieved chunks.
  \end{itemize}

  \textbf{Learner-update extraction.} \texttt{learner\_analyzer.py}
  runs another structured-output LLM call --- output schema:
  \texttt{covered}, \texttt{demonstrated\_understanding}, \texttt{showed\_confusion}.
  These become the \texttt{LearnerUpdates} returned to the backend.
\end{frame}

\begin{frame}{Quiz Gen --- Step 1: Bloom's Taxonomy Extraction}
  \begin{block}{What it produces}
    \scriptsize
    \texttt{services/concept\_extractor.py} prompts the LLM with a
    Pydantic schema \texttt{ExtractedConcepts}: lists of
    \texttt{ConceptCard}s grouped by Bloom's level
    \texttt{\{remember, understand, apply, analyze\}}. Each card carries
    \texttt{name}, \texttt{description}, and the
    \texttt{source\_chunk\_ids} it came from.
  \end{block}

  \begin{exampleblock}{Bloom's pyramid (used downstream)}
    \scriptsize
    \begin{tabular}{l p{7cm}}
      \texttt{remember}   & Factual recall (terms, dates) \\
      \texttt{understand} & Paraphrase, summarize \\
      \texttt{apply}      & Use a procedure in a new situation \\
      \texttt{analyze}    & Compare, contrast, decompose \\
    \end{tabular}
  \end{exampleblock}

  \scriptsize
  Mapping concept $\to$ source chunk IDs is preserved so the
  question generator can quote the right passage when scripting a
  question.
\end{frame}

\begin{frame}{Quiz Gen --- Step 2: Learner-Aware Difficulty Planner}
  \scriptsize
  \texttt{services/difficulty\_adapter.py} allocates slots over
  Bloom's levels using the \texttt{LearnerState}:

  \begin{tcolorbox}[colback=eduLightOrange, colframe=eduOrange, title=Slot allocation]
    \scriptsize
    \begin{tabular}{p{2.8cm} p{2.2cm} p{6cm}}
      \toprule
      \textbf{Slot type} & \textbf{Share} & \textbf{Purpose} \\
      \midrule
      Struggling-concept & 60\,\%        & Reinforce weak areas at \texttt{remember/understand}. \\
      Coverage           & 30\,\%        & New breadth across the material. \\
      Stretch            & 10\,\%        & Push understood concepts to \texttt{apply/analyze}. \\
      \bottomrule
    \end{tabular}
  \end{tcolorbox}

  Each slot also carries a \texttt{QuestionKind} (\texttt{mcq},
  \texttt{true\_false}, \texttt{fill\_blank}). The frontend can
  restrict allowed kinds via the \texttt{options.question\_types} field;
  the planner respects those filters.
\end{frame}

\begin{frame}{Quiz Gen --- Steps 3--5: Generate, Enhance, Validate}
  \scriptsize
  \begin{block}{3. Generate (structured)}
    \texttt{services/question\_generator.py} runs one LLM call per slot
    with the Pydantic schema for that question kind (MCQ / TF / FillBlank).
    Each question carries \texttt{bloom\_level}, \texttt{concept},
    \texttt{explanation}, and \texttt{source\_chunk\_id} so the UI can
    cite the grounding chunk.
  \end{block}

  \begin{block}{4. Enhance distractors (MCQs only)}
    \texttt{services/distractor\_engine.py} re-prompts the LLM for
    \textit{plausibly wrong} options drawn from the same context
    chunks. Better distractors $\to$ higher pedagogical value.
  \end{block}

  \begin{block}{5. Validate}
    \texttt{services/quality\_validator.py} performs:
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item Pydantic schema validation per question.
      \item Deduplication on normalized question text.
      \item Concept-label repair (e.g.\ \texttt{intent} $\to$
            ``Definition of Intent'').
      \item Bloom-distribution check.
      \item \textbf{Top-up}: if survivors $<$ target, synthesize
            grounded \texttt{TrueFalse} questions ``The source X covers
            concept Y'' to guarantee a full-length quiz.
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Quiz Gen --- Steps 6--7: Intro \& Artifact}
  \scriptsize
  \begin{block}{Adaptive intro (teacher voice)}
    \texttt{adaptive\_guidance.txt} prompt is rendered with the
    learner's struggling \& understood lists, the total question count,
    and the Bloom mix. Produces messages like:
    \emph{``Let's test what you know! I've focused on things we've been
    working on: \texttt{intents}, \texttt{activity lifecycle}. Ready?''}
  \end{block}

  \begin{block}{Artifact export}
    The final \texttt{QuizOutput} (intro + questions + bloom
    distribution) is serialised to JSON, uploaded to MinIO via
    \texttt{ArtifactStore}, and exposed via a presigned URL inside the
    \texttt{GeneratorArtifact} returned to the frontend. The frontend's
    \texttt{QuizRenderer} consumes that JSON to render interactive
    radios / inputs / score calculation.
  \end{block}

  \scriptsize
  The whole run streams SSE \texttt{progress} events:
  \texttt{extracting\_concepts} $\to$ \texttt{concepts\_extracted} $\to$
  \texttt{planned} $\to$ \texttt{generated} $\to$
  \texttt{distractors\_enhanced} $\to$ \texttt{validated} $\to$
  \texttt{done}, so the UI can show real-time progress.
\end{frame}

\subsection*{Mind Map Generator}
% =====================================================================

\begin{frame}{Mind Map Generator (\texttt{mindmap\_gen :8008})}
  \scriptsize
  \textbf{Mission.} Birds-eye view of the entire uploaded corpus.
  Central topic + 3--7 main branches + 2--5 sub-branches + leaves.

  \begin{block}{Key architectural decision}
    Unlike LLM-generated raw Mermaid (error-prone), this generator asks the
    LLM for a \textbf{structured hierarchy} (recursive Pydantic
    \texttt{MindMapNode}) and then \textbf{programmatically compiles} the
    tree into Mermaid markdown $\Rightarrow$ \textit{guaranteed-valid
    output}.
  \end{block}

  \begin{exampleblock}{Multi-strategy pipeline with cascading fallbacks}
    \scriptsize
    \begin{enumerate}
      \setlength\itemsep{0pt}
      \item \textbf{Module pack} build (if backend provided
            \texttt{mindmap\_module\_pack} chunks) --- no LLM, purely
            structural.
      \item Otherwise: per-batch LLM outline extraction $\to$
            \textbf{course synthesis} LLM call to merge partial outlines.
      \item Whole-course outline LLM fallback.
      \item Parsed-Markdown structural fallback.
      \item Theme-by-theme legacy expansion (last resort).
    \end{enumerate}
    Each stage emits SSE \texttt{progress} events with sub-stages and
    keep-alives (every 1\,s while a long LLM call runs, until
    \texttt{LLM\_CALL\_TIMEOUT\_S}).
  \end{exampleblock}

  \scriptsize
  Post-processing: \texttt{balancer.balance(\dots)} caps the node count
  and rebalances depth; \texttt{markdown\_compiler.compile(\dots)}
  produces Markmap-compatible markdown; \texttt{html\_renderer.render}
  produces an interactive standalone HTML (Mermaid.js) artifact.
\end{frame}

\section{LearnerState - Adaptive Tutoring}
% =====================================================================

\begin{frame}{LearnerState --- What Makes the Teacher Adaptive}
  \scriptsize
  Stored in Postgres (\texttt{LearnerState(conversation\_id PK, state\_json, updated\_at)}).
  Loaded by the backend on every dispatch, passed in
  \texttt{GeneratorInput.learner\_state}, mutated by
  \texttt{learner\_updates} returned by the generator.

  \begin{block}{Schema (\texttt{teacherlm\_core/schemas/learner\_state.py})}
    \scriptsize
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item \texttt{understood\_concepts: list[str]} --- mastery $\geq 0.7$.
      \item \texttt{struggling\_concepts: list[str]} --- mastery $\leq 0.3$.
      \item \texttt{mastery\_scores: dict[str, float]} --- per-concept $\in [0,1]$.
      \item \texttt{session\_turns: int} --- conversation length.
      \item \texttt{turns\_since\_progress: int} --- detects ``stuck'' students.
    \end{itemize}
  \end{block}

  \begin{exampleblock}{Mastery update math (in \texttt{learner\_tracker.py})}
    \scriptsize
    On each generator return:
    \begin{itemize}
      \setlength\itemsep{0pt}
      \item For $c \in$ \texttt{concepts\_demonstrated}:\
            $m_c \leftarrow m_c + 0.2\,(1 - m_c)$.
      \item For $c \in$ \texttt{concepts\_struggled}:\
            $m_c \leftarrow 0.7 \cdot m_c$.
      \item Thresholds: $m_c \geq 0.7$ promotes to \emph{understood};
            $m_c \leq 0.3$ flags \emph{struggling}.
    \end{itemize}
  \end{exampleblock}

  \scriptsize
  This state \textbf{loops back} into every subsequent
  generation --- the teacher changes tone, quizzes focus on weak areas,
  introductions reference specific past struggles.
\end{frame}

\section{Key Design Decisions}
% =====================================================================

\begin{frame}{Why Local LLMs (Ollama)?}
  \begin{itemize}
    \item \textbf{Privacy.} Student course materials never leave the
          machine.
    \item \textbf{Cost.} No per-token API charges --- unlimited usage on
          provisioned hardware.
    \item \textbf{Reproducibility.} Fully transparent, no vendor lock-in.
    \item \textbf{Offline.} Works after the models are pulled.
    \item \textbf{Structured outputs.} Ollama's native \texttt{format=}
          accepts a JSON schema --- Pydantic V2's
          \texttt{model\_json\_schema()} plugs in directly, replacing
          the whole LangChain ``output parser'' stack with one
          \texttt{ollama.chat(\dots, format=schema)} call plus a
          retry-on-validation-error wrapper.
  \end{itemize}
\end{frame}

\begin{frame}{Why \emph{Not} LangChain / LangGraph?}
  \begin{itemize}
    \item Pydantic V1 deprecation warnings break strict mode on
          Python 3.14.
    \item Heavy abstraction for our actual surface area --- a thin
          \texttt{OllamaClient} + an \texttt{HTTPX} dispatcher already
          cover everything.
    \item Direct LLM calls = direct debugging. Each call is a single
          stack frame, not five chained Runnables.
    \item No surprises around prompt mutation, memory back-stores, or
          tool-call autoflow. The pipelines are explicit step-functions.
  \end{itemize}

  \begin{alertblock}{Other non-negotiables (root \texttt{CLAUDE.md})}
    \scriptsize
    No \texttt{llama-parse} / \texttt{llama-cloud-services} (deprecated
    May 2026). Use \texttt{llama-cloud} $\geq 1.0$ instead. Pydantic V2
    only. FastAPI $\geq 0.135$. \texttt{fastembed} where avoidable
    instead of \texttt{sentence-transformers}.
  \end{alertblock}
\end{frame}

\begin{frame}{Why Microservice-per-Generator?}
  \begin{itemize}
    \item \textbf{Independent scaling.} \texttt{teacher\_gen} streams
          tokens and wants low latency; \texttt{quiz\_gen} runs many
          sequential LLM calls and tolerates much higher latency.
    \item \textbf{Independent deployment.} Ship a new quiz pipeline
          without restarting the chat service.
    \item \textbf{Independent models.} Each generator owns its own
          Ollama client and can pick a different model for different
          steps (teacher uses 3 model configurations in one pipeline).
    \item \textbf{Plug-in extensibility.} Adding a generator =
          1 microservice + 1 JSON entry. No backend changes.
    \item \textbf{Failure isolation.} A panicking quiz generator
          doesn't take down chat.
  \end{itemize}
\end{frame}

\begin{frame}{Why RRF (and Why Hybrid)?}
  \begin{itemize}
    \item Dense models miss \textbf{exact matches} on domain terms
          (RMSE, nDCG, Henderson--Hasselbalch).
    \item BM25 misses \textbf{paraphrase} (``causes of photosynthesis''
          $\not\rightarrow$ ``light-dependent reactions'').
    \item RRF requires \textbf{no normalisation} between the two scoring
          scales --- just ranks. It's parameter-free and robust
          (Cormack et al., 2009).
    \item Re-ranking with a cross-encoder afterwards gives precision
          while keeping recall cheap.
  \end{itemize}

  \begin{exampleblock}{Why 5 retrieval modes?}
    \scriptsize
    Different output types have different information needs ---
    \textbf{the same vector store is queried under five different
    lenses}. Chat needs the most relevant chunks; quizzes need broad
    coverage; podcasts need narrative shape; reports need topical
    clusters; charts need entity-dense chunks.
  \end{exampleblock}
\end{frame}

\begin{frame}{Why fastembed (not sentence-transformers)?}
  \begin{itemize}
    \item Smaller dependency footprint --- no PyTorch.
    \item CPU-optimised via ONNX runtime --- a real win on machines
          without a GPU.
    \item Single library covers both bi-encoder (\texttt{TextEmbedding})
          and cross-encoder (\texttt{TextCrossEncoder}) --- no second
          dependency for reranking.
    \item Models warmed into a Docker-managed volume (\texttt{fastembed\_cache})
          so subsequent rebuilds are fast.
  \end{itemize}
\end{frame}

\begin{frame}{Why LlamaCloud (and not local PDF extraction)?}
  \begin{itemize}
    \item Course PDFs are full of \textbf{multi-column text}, tables,
          embedded images, equations, headers/footers --- everything
          local parsers struggle with.
    \item LlamaCloud (the \texttt{llama-cloud $\geq 1.0$} SDK, not the
          deprecated \texttt{llama-parse}) returns clean \textbf{Markdown}
          that preserves hierarchy.
    \item Better Markdown $\Rightarrow$ better section extraction
          $\Rightarrow$ better chunks $\Rightarrow$ better retrieval.
          Garbage in, hallucinations out.
    \item All other moving parts (embedding, retrieval, generation)
          remain fully local --- only parsing is offloaded.
  \end{itemize}
\end{frame}

\section{Conclusion}
% =====================================================================

\begin{frame}{Summary --- What Makes TeacherLM Different}
  \begin{itemize}
    \item \textbf{Strict grounding} as a system property, enforced by
          off-topic guards, groundedness scoring, and refusal paths ---
          not just a prompt suggestion.
    \item \textbf{Hybrid retrieval} (dense + BM25 + RRF) plus
          \textbf{cross-encoder reranking}, plus \textbf{five retrieval
          modes} chosen per output type.
    \item \textbf{Multi-agent} architecture: every output type is its
          own FastAPI microservice with its own pipeline, prompts, and
          models, registered via one JSON file.
    \item \textbf{Adaptive} via a per-conversation \texttt{LearnerState}
          updated after every turn and fed back into every generator.
    \item \textbf{Fully local} inference (Ollama + fastembed) and
          \textbf{strict modern Python} (3.14, Pydantic V2, FastAPI 0.135,
          no LangChain, no llama-parse).
    \item \textbf{Operational maturity}: SSE everywhere, crash-safe
          ingestion worker, per-conversation Qdrant collections, MinIO
          artifacts, Alembic migrations.
  \end{itemize}
\end{frame}
