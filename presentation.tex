\section{Pipeline RAG avanc\'{e}}

\begin{frame}{Pipeline d'ingestion}
  \textbf{Upload $\rightarrow$ Parse $\rightarrow$ Chunk $\rightarrow$ Embed $\rightarrow$ Ready}
  
  \vspace{0.3cm}
  \begin{itemize}
    \item Worker asynchrone (arq)
    \item LlamaCloud pour parsing PDF
    \item fastembed pour embeddings
    \item Stockage dans Qdrant
    \item Progression en temps r\'{e}el
  \end{itemize}
\end{frame}

\begin{frame}{Retrieval hybride : 5 modes}
  \small
  \begin{tabular}{lp{4cm}c}
    \toprule
    \textbf{Mode} & \textbf{Description} & \textbf{k} \\
    \midrule
    semantic\_topk & Dense + BM25 + RRF & 12 \\
    coverage\_broad & MMR (diversit\'{e}) & 15 \\
    topic\_clusters & K-means sur embeddings & Var \\
    metadata\_filtered & Filtrage attributs & 10 \\
    contextual\_expansion & Chunks voisins & Var \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}{Reciprocal Rank Fusion (RRF)}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{BM25 (sparse):}
      \begin{itemize}\small
        \item Tokenisation
        \item Scoring BM25Okapi
        \item k = 25 candidats
        \item Excellent pour termes exacts
      \end{itemize}
      
      \vspace{0.3cm}
      \textbf{S\'{e}mantique (dense):}
      \begin{itemize}\small
        \item Embeddings bge-small
        \item Similarit\'{e} cosinus
        \item k = 30+ candidats
        \item Excellent pour paraphrases
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Formule RRF:}
      \[
        \text{RRF}(d) = \sum_{r} \frac{1}{k + \text{rank}_r(d)}
      \]
      
      \vspace{0.3cm}
      \textbf{Avantages:}
      \begin{itemize}\small
        \item Sans param\`{e}tres
        \item Robuste
        \item Combine forces compl\'{e}mentaires
        \item Performance sup\'{e}rieure
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}
